226      private static Session getSession(HttpExchange ex) {
227          String cookie = ex.getRequestHeaders().getFirst("Cookie");
228          if (cookie == null) return null;
229          for (String part : cookie.split(";")) {
230              String[] kv = part.trim().split("=",2);
231              if (kv.length==2 && kv[0].equals("SESSION")) {
232                  return sessions.get(kv[1]);
233              }
234          }
235          return null;
236      }
237      private static void setSession(HttpExchange ex, String username, String role) {
238          String token = Long.toHexString(random.nextLong()) + Long.toHexString(System.nanoTime());
239          Session s = new Session(); s.username = username; s.role = role;
240          sessions.put(token, s);
241          ex.getResponseHeaders().add("Set-Cookie", "SESSION="+token+"; HttpOnly; Path=/");
242      }
243      private static void clearSession(HttpExchange ex) {
244          String cookie = ex.getRequestHeaders().getFirst("Cookie");
245          if (cookie != null) {
246              for (String part : cookie.split(";")) {
247                  String[] kv = part.trim().split("=",2);
248                  if (kv.length==2 && kv[0].equals("SESSION")) {
249                      sessions.remove(kv[1]);
250                  }
251              }
252          }
253          ex.getResponseHeaders().add("Set-Cookie","SESSION=deleted; Max-Age=0; Path=/");
254      }

321      private static Map<String,String> parseForm(String data){
322          Map<String,String> m = new HashMap<>();
323          if (data==null || data.isEmpty()) return m;
324          for (String kv : data.split("&")){
325              String[] parts = kv.split("=",2);
326              String k = URLDecoder.decode(parts[0], StandardCharsets.UTF_8);
327              String v = parts.length>1 ? URLDecoder.decode(parts[1], StandardCharsets.UTF_8) : "";
328              m.put(k,v);
329          }
330          return m;
331      }
